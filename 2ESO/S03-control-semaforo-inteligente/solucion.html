S03 — Control de semáforo inteligente

2º ESO · Tinkercad · Emergencia con parpadeo + amarillo fijo

El reto era…

Cuando se detecte emergencia: amarillo parpadeando rápido (0,25 s ON + 0,25 s OFF) durante 4 segundos exactos (8 parpadeos completos), seguido de amarillo fijo durante 4 segundos más. Después vuelve al ciclo normal.

Arduino C++ 
void setup() {
    pinMode(13, OUTPUT);  // LED verde
    pinMode(12, OUTPUT);  // LED amarillo
    pinMode(11, OUTPUT);  // LED rojo
    pinMode(6,  INPUT);   // interruptor
    pinMode(7,  INPUT);   // pulsador
}

void loop() {
    if (digitalRead(6) == HIGH) {
        bool emergencia = false;

        // Verde 1 s
        digitalWrite(13, HIGH);
        if (digitalRead(7) == HIGH) emergencia = true;
        delay(1000);
        digitalWrite(13, LOW);

        // Amarillo 1 s
        digitalWrite(12, HIGH);
        if (digitalRead(7) == HIGH) emergencia = true;
        delay(1000);
        digitalWrite(12, LOW);

        // Rojo 1 s
        digitalWrite(11, HIGH);
        if (digitalRead(7) == HIGH) emergencia = true;
        delay(1000);
        digitalWrite(11, LOW);

        if (emergencia == true) {
            // 8 parpadeos rapidos = 4 s exactos
            digitalWrite(12, HIGH); delay(250);
            digitalWrite(12, LOW);  delay(250);
            digitalWrite(12, HIGH); delay(250);
            digitalWrite(12, LOW);  delay(250);
            digitalWrite(12, HIGH); delay(250);
            digitalWrite(12, LOW);  delay(250);
            digitalWrite(12, HIGH); delay(250);
            digitalWrite(12, LOW);  delay(250);
            digitalWrite(12, HIGH); delay(250);
            digitalWrite(12, LOW);  delay(250);
            digitalWrite(12, HIGH); delay(250);
            digitalWrite(12, LOW);  delay(250);
            digitalWrite(12, HIGH); delay(250);
            digitalWrite(12, LOW);  delay(250);
            digitalWrite(12, HIGH); delay(250);
            digitalWrite(12, LOW);  delay(250);

            // Amarillo fijo 4 s
            digitalWrite(12, HIGH);
            delay(4000);
            digitalWrite(12, LOW);
        }

    } else {
        digitalWrite(13, LOW);
        digitalWrite(12, LOW);
        digitalWrite(11, LOW);
    }
}
